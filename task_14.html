1 Filter Out Negative Numbers

function filterNegativeNumbers(numbers) { return numbers.filter(number => number >= 0); } const numbers = [5, -1, 10, -7, 0, 42, -23, 15]; const positiveNumbers = filterNegativeNumbers(numbers); console.log("Original array:", numbers); console.log("Filtered array:", positiveNumbers);

2 Difference from 13

function differenceFrom13(number) { const difference = number - 13; if (difference > 0) { return 2 * Math.abs(difference); } else { return Math.abs(difference); } } console.log("Difference from 13 for 10:", differenceFrom13(10)); console.log("Difference from 13 for 20:", differenceFrom13(20));

3 Fibonacci Series

function generateFibonacci(n) { const fibSeries = []; if (n <= 0) return fibSeries; fibSeries.push(0); if (n === 1) return fibSeries; fibSeries.push(1); for (let i = 2; i < n; i++) { fibSeries.push(fibSeries[i-1] + fibSeries[i-2]); } return fibSeries; } const fibNumbers = generateFibonacci(10); console.log("Fibonacci series (10 numbers):", fibNumbers);

4 Sum of Multiples of 3 and 5 Under 1000

function sumMultiples3And5(limit) { let sum = 0; for (let i = 0; i < limit; i++) { if (i % 3 === 0 || i % 5 === 0) { sum += i; } } return sum; } const sum = sumMultiples3And5(1000); console.log("Sum of multiples of 3 and 5 under 1000:", sum);

5 Cube of a Number Using Function Object
