1 Sort a JavaScript Array

function sortArray(array) { // Creating a copy to avoid modifying the original array const sortedArray = [...array]; // Sort the array (default is ascending order) sortedArray.sort((a, b) => a - b); return sortedArray; } const numbers = [34, 7, 23, 32, 5, 62]; const sortedNumbers = sortArray(numbers); console.log("Original array:", numbers); console.log("Sorted array:", sortedNumbers);

2 Concatenate Two Arrays

function concatenateArrays(array1, array2) { // Create a new array by concatenating both arrays const newArray = [...array1, ...array2]; return newArray; } const firstArray = [1, 2, 3]; const secondArray = [4, 5, 6]; const concatenatedArray = concatenateArrays(firstArray, secondArray); console.log("First array:", firstArray); console.log("Second array:", secondArray); console.log("Concatenated array:", concatenatedArray);

3 Remove First Element from Array

function removeFirstElement(array) { // Create a copy to avoid modifying the original array const newArray = [...array]; // Remove the first element newArray.shift(); return newArray; } const originalArray = ["apple", "banana", "cherry", "date", "elderberry"]; const modifiedArray = removeFirstElement(originalArray); console.log("Original array:", originalArray); console.log("Array after removing first element:", modifiedArray);
